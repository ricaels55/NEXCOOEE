// Sistema completo sem senha
let peopleDatabase = [];
let currentSearchTerm = '';

function initHackerRain() {
    const bg = document.getElementById('matrixBg');
    if (!bg) return;
    bg.innerHTML = '';
    for (let i = 0; i < 40; i++) {
        const rain = document.createElement('div');
        rain.className = 'rain';
        const chars = '01アイウエオカキクケコサシスソタチツテトナニヌネノハヒフヘホマミムメモ';
        let txt = '';
        for (let j = 0; j < 25; j++) txt += chars[Math.floor(Math.random() * chars.length)];
        rain.textContent = txt;
        rain.style.left = Math.random() * 100 + '%';
        rain.style.fontSize = (Math.random() * 18 + 12) + 'px';
        rain.style.animationDuration = (Math.random() * 5 + 3) + 's';
        rain.style.animationDelay = Math.random() * 10 + 's';
        rain.style.opacity = Math.random() * 0.5 + 0.2;
        bg.appendChild(rain);
    }
}

function saveData() {
    localStorage.setItem('nexcooee_people_v2', JSON.stringify(peopleDatabase));
}

function loadData() {
    const stored = localStorage.getItem('nexcooee_people_v2');
    if (stored) {
        try { peopleDatabase = JSON.parse(stored); } catch(e) { peopleDatabase = []; }
    }
    if (!peopleDatabase || !peopleDatabase.length) {
        peopleDatabase = [
            { id: '1', name: 'Alice Kessler', email: 'alice@darkweb.com', phone: '+1 555-0987', location: 'Brooklyn, NY', createdAt: Date.now() },
            { id: '2', name: 'Marcus ZeroDay', email: 'marcus@proton.me', phone: '+55 98765-4321', location: 'São Paulo, SP', createdAt: Date.now() }
        ];
        saveData();
    }
    renderList();
}

function updateCount() {
    const span = document.getElementById('recordCount');
    if (span) span.innerText = peopleDatabase.length;
}

function addPerson(name, email, phone, location) {
    if (!name.trim()) { alert('⚠️ Nome é obrigatório!'); return false; }
    const newPerson = {
        id: Date.now().toString() + Math.random().toString(36).substr(2, 4),
        name: name.trim(),
        email: email.trim(),
        phone: phone.trim(),
        location: location.trim() || 'Desconhecido',
        createdAt: Date.now()
    };
    peopleDatabase.unshift(newPerson);
    saveData();
    renderList();
    showStatus('✅ ' + newPerson.name + ' adicionado!');
    return true;
}

function deletePerson(id) {
    const person = peopleDatabase.find(p => p.id === id);
    if (person && confirm('Remover ' + person.name + ' permanentemente?')) {
        peopleDatabase = peopleDatabase.filter(p => p.id !== id);
        saveData();
        renderList();
        showStatus('🗑️ ' + person.name + ' removido!');
    }
}

function getFilteredPeople() {
    if (!currentSearchTerm.trim()) return [...peopleDatabase];
    const term = currentSearchTerm.toLowerCase();
    return peopleDatabase.filter(p => 
        p.name.toLowerCase().includes(term) ||
        p.email.toLowerCase().includes(term) ||
        p.location.toLowerCase().includes(term) ||
        p.phone.includes(term)
    );
}

function highlightText(text) {
    if (!currentSearchTerm.trim() || !text) return text || '';
    const term = currentSearchTerm.toLowerCase();
    const lowerText = text.toLowerCase();
    if (!lowerText.includes(term)) return text;
    const regex = new RegExp(`(${term.replace(/[.*+?^${}()|[\]\\]/g, '\\$&')})`, 'gi');
    return text.replace(regex, '<span style="background:#1effbc30; padding:0 2px; border-radius:6px;">$1</span>');
}

function renderList() {
    const grid = document.getElementById('peopleGrid');
    if (!grid) return;
    const filtered = getFilteredPeople();
    if (filtered.length === 0) {
        grid.innerHTML = '<div class="empty-msg"><i class="fas fa-user-slash"></i> Nenhum registro encontrado.<br>Adicione novos registros ou limpe o filtro.</div>';
        updateCount();
        return;
    }
    const html = filtered.map(person => `
        <div class="data-card" data-id="${person.id}">
            <div style="display:flex;justify-content:space-between;align-items:baseline;">
                <span class="card-name">${highlightText(person.name)}</span>
                <span class="badge" style="font-size:0.65rem;">${person.id.slice(-6)}</span>
            </div>
            <div class="info-row"><i class="fas fa-envelope"></i> ${highlightText(person.email) || '—'}</div>
            <div class="info-row"><i class="fas fa-phone-alt"></i> ${highlightText(person.phone) || '—'}</div>
            <div class="info-row"><i class="fas fa-map-marker-alt"></i> ${highlightText(person.location) || '—'}</div>
            <div class="badge"><i class="fas fa-clock"></i> ${new Date(person.createdAt).toLocaleString()}</div>
            <div class="delete-icon" data-del="${person.id}"><i class="fas fa-trash-alt"></i></div>
        </div>
    `).join('');
    grid.innerHTML = html;
    updateCount();
    document.querySelectorAll('.delete-icon').forEach(icon => {
        icon.addEventListener('click', (e) => {
            e.stopPropagation();
            deletePerson(icon.getAttribute('data-del'));
        });
    });
    document.querySelectorAll('.data-card').forEach(card => {
        card.addEventListener('click', (e) => {
            if (e.target.closest('.delete-icon')) return;
            const person = peopleDatabase.find(p => p.id === card.getAttribute('data-id'));
            if (person) {
                alert(`🔎 PERFIL\n\n${person.name}\n📧 ${person.email || '—'}\n📞 ${person.phone || '—'}\n📍 ${person.location || '—'}`);
            }
        });
    });
}

function showStatus(msg) {
    const statusDiv = document.getElementById('addStatus');
    if (statusDiv) {
        statusDiv.innerText = msg;
        setTimeout(() => { statusDiv.innerText = ''; }, 3000);
    }
}

// Backup e Restore
function fazerBackup() {
    const dados = { pessoas: peopleDatabase, dataBackup: new Date().toISOString() };
    const blob = new Blob([JSON.stringify(dados, null, 2)], { type: 'application/json' });
    const a = document.createElement('a');
    a.href = URL.createObjectURL(blob);
    a.download = `nexcooee_backup_${Date.now()}.json`;
    a.click();
    showStatus('✅ Backup salvo!');
}

function restaurarBackup(file) {
    const reader = new FileReader();
    reader.onload = function(e) {
        try {
            const dados = JSON.parse(e.target.result);
            if (dados.pessoas && Array.isArray(dados.pessoas)) {
                peopleDatabase = dados.pessoas;
                saveData();
                renderList();
                showStatus(`✅ Backup restaurado! ${peopleDatabase.length} registros.`);
            } else { showStatus('❌ Arquivo inválido!'); }
        } catch(error) { showStatus('❌ Erro ao ler arquivo!'); }
    };
    reader.readAsText(file);
}

// Eventos
document.getElementById('addBtn')?.addEventListener('click', () => {
    addPerson(
        document.getElementById('fullName').value,
        document.getElementById('personEmail').value,
        document.getElementById('personPhone').value,
        document.getElementById('personLocation').value
    );
    document.getElementById('fullName').value = '';
    document.getElementById('personEmail').value = '';
    document.getElementById('personPhone').value = '';
    document.getElementById('personLocation').value = '';
});
document.getElementById('searchInput')?.addEventListener('input', (e) => {
    currentSearchTerm = e.target.value;
    renderList();
});
document.getElementById('clearSearchBtn')?.addEventListener('click', () => {
    document.getElementById('searchInput').value = '';
    currentSearchTerm = '';
    renderList();
});
document.getElementById('backupBtn')?.addEventListener('click', fazerBackup);
document.getElementById('restoreBtn')?.addEventListener('click', () => {
    const input = document.createElement('input');
    input.type = 'file';
    input.accept = '.json';
    input.onchange = (e) => { if (e.target.files[0]) restaurarBackup(e.target.files[0]); };
    input.click();
});

initHackerRain();
loadData();

    const bg = document.getElementById('matrixBg');

                const chars = '01アイウエオカキクケコサシスソ';

                for (let j = 0; j < r.textContent.length; j++) newTxt += chars[Math.floor(Math.random() * chars.length)];
                r.textContent = newTxt;
            }
        }
    }
}, 1500);
